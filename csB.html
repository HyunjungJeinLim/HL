CSS Cheat Sheet

Property Example
color color: blue;
font-size font-size: 16px;
font-family font-family: Arial, sans-serif;
background-color background-color: #f2f2f2;
padding padding: 10px;
margin margin: 20px;
border border: 1px solid #ddd;
text-align text-align: center;

Selector Example
ID #SpecialParagraph {
color: red;
}
Class .allIntroductionParagraphs {(or_ p.pink {)
color: pink;
}
Grouping #paragraph1 {
font-size: 40px;
font-weight: bold;
color: green;
}
Universal * {
margin: 0;
padding: 0;
}
Pseudo a:link { background-color: gold; color: green; }
Attribute /* tag with a title attribute */
a[title] {
color: blue;
}

/* tag with an href matching "https://google.com" */
a[href="https://google.com"] {
color: magenta;
}

/* tag with an href containing "google" */
a[href*="google"] {
font-style: italic;
}
Descendant ul.fruits li a {
text-decoration: none;
}
Adjacent h1 + p {
color: green;
font-size: 20px;
text-decoration: underline;
}
Child ul > li {
list-style: disc;
}
:nth-child() /* Selects the second <li> element in a list */
li:nth-child(2) {
color: blue;
}

/* Selects every fourth element among any group of siblings */
:nth-child(4n) {
text-decoration: underline;
}

/* Selects every even element among any group of siblings */
ul:nth-child(even) {
color: gray;
}
